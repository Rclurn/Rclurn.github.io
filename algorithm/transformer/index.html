class PositionEmbedding(nn.Module):
    def __init__(self, dim, base=10000, max_seq_len=512):
        super().__init__()
        self.dim = dim
        pe = torch.zeros(max_seq_len, dim)
        for pos in range(max_seq_len):
            for i in range(0, dim, 2):
                pe[pos][i] = math.sin(pos / (base ** (2 * i / self.dim)))
                pe[pos][i + 1] = math.cos(pos / (base ** (2 * (i + 1) / self.dim)))
        pe = pe.unsqueeze(dim=0)
        self.register_buffer('pe', pe)
    def forward(self, x):
        x = x * math.sqrt(self.dim) + self.pe[:, :s.shape[1], :].to(x.device)
        return x

class RotaryEmbedding(nn.Module):
    def __init__(self, dim, base=10000):
        super().__init__()
        self.dim = dim
        inv_freq = 1.0 / (base ** (torch.range(0, dim, 2) / dim))
        self.register_buffer('inv_freq', inv_freq, persistent=False)
    @torch.no_grad()
    def forward(self, x, position_ids):
        expanded_inv_freq = self.inv_freq[None, :, None].expand(position_ids.shape[0], -1, 1)
        expanded_position = position_ids[:, None, :]
        with torch.autocast():
            freqs = (expanded_inv_freq @ expanded_position).transpose(1, 2)
            emb = torch.concat([freqs, freqs], dim=-1)
            cos = emb.cos()
            sin = emb.sin()
        return cos, sin
def rotate_half(x):
    x1 = x[..., :x.shape[-1] // 2]
    x2 = x[..., x.shape[-1] // 2:]
    return torch.concat([-x2, x1], dim=-1)
def apply_rotate_to_emb(q, k, cos, sin, unsqueezed_dim=1):
    cos = cos.unsqueeze(unsqueezed_dim)
    sin = sin.unsqueeze(unsqueezed_dim)
    embed_q = (q * cos) + (rotate_half(q) * sin)
    embed_k = (k * sin) + (rotate_half(k) * cos)
    return embed_q, embed_k

def make_custom_mask(attention_mask):
    bs, seq_len = attention_mask.size()
    mask = torch.zeros(bs, seq_len, seq_len)
    mask = mask + attention_mask[:, None, :]
    mask = (1 - mask) * torch.finfo(torch.float16).min
    return mask[:, None, :, :]
def make_casual_mask(attention_mask):
    bs, seq_len = attention_mask.size()
    mask = torch.full((seq_len, seq_len), torch.finfo(torch.float16).min)
    mask_cond = torch.arange(seq_len)
    mask.masked_fill_(mask_cond < (mask_cond + 1).view(seq_len, 1), 0)
    return mask[None, None, :, :].expand(bs, -1, seq_len, seq_len)
class MultiHeadAttention(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.config = config
        self.Wq = nn.Linear(config.hidden_size, config.head_num * config.head_size)
        self.Wk = nn.Linear(config.hidden_size, config.kv_head * config.head_size)
        self.Wv = nn.Linear(config.hidden_size, config.kv_head * config.head_size)
        self.Wo = nn.Linear(config.head_num * config.head_size, config.hidden_size)
        self.attn_drop = nn.Dropout(config.attn_drop_prob)
        self.state_drop = nn.Dropout(config.state_drop_prob)
    def forward(self, hidden_states, attention_mask):
        bs, seq_len, hidden_size = hidden_states.size()
        Q = self.Wq(hidden_states).view(bs, seq_len, self.config.head_num, self.config.head_size).permute(0, 2, 1, 3)
        K = self.Wk(hidden_states).view(bs, seq_len, self.config.kv_head, self.config.head_size).permute(0, 2, 1, 3)
        V = self.Wv(hidden_states).view(bs, seq_len, self.config.kv_head, self.config.head_size).permute(0, 2, 1, 3)
        if self.config.kv_group != 1:
            K = K[:, :, None, :, :].expand(bs, self.config.kv_head, self.config.kv_group, seq_len, self.config.head_size).reshape(bs, self.config.head_num, seq_len, hidden-size)
            V = V[:, :, None, :, :].expand(bs, self.config.kv_head, self.config.kv_group, seq_len, self.config.head_size).reshape(bs, self.config.head_num, seq_len, hidden-size)
        attention_score = torch.matmul(Q, K.transpose(2, 3)) / math.sqrt(hidden_size)

        O = self.state_drop(self.Wo(O))

class LayerNorm(nn.Module):
    def __init__(self, hidden_size, eps=1e-6):
        super().__init__()
        self.eps = eps
        self.gamma = nn.Parameter(torch.ones(hidden_size))
        self.beta = nn.Parameter(torch.zeros(hidden_size))
    def forward(self, x):
        mean = x.mean(dim=-1, keepdim=True)
        var = x.var(dim=-1, keepdim=True)
        normlized_x = (x - mean) / (var + self.eps)
        return self.gamma * norlized_x + self.beta

class RmsNorm(nn.Module):
    def __init__(self, hidden_size, eps=1e-6):
        super().__init__()
        self.eps = eps
        self.weight = nn.Parameter(torch.ones(hidden_size))
    def forward(self, x):
        root_mean = x.pow(2).mean(dim=-1, keepdim=True)
        x = x * torch.rsqrt(root_mean + self.eps)
        return x * self.weight
