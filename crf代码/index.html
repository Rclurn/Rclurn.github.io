# 计算真实路径得分
def score_target_sentence(self, emit_matrixs, tags, attention_mask):
    seq_len, batch_size, tag_size = emit_matrixs.size()

    scores = emit_matrixs.new_zeros(batch_size) # [batch_size]
    # cat([1, batch_size], [seq_len, batch_size]) -> [seq_len + 1, batch_size]
    # 在标注序列的前面加上START_TAG标注，并将其得分设置为0
    tags = torch.cat([tags.new_full((1, batch_size), fill_value=tags2label['START_TAG']), tags], 0)
    # 在时间序列维度上遍历发射矩阵
    for i, emit_score in enumerate(emit_matrixs):
        # 获得当前时间步的batch中每个字的发射矩阵得分
        emit_score = torch.stack([f[next_tag] for f, next_tag in zip(emit_score, tags[i + 1])])
        # 获得当前时间步到下一个时间步的转移矩阵得分
        transition_score = torch.stack([self.transition[tags[i + 1, b], tags[i, b]] for b in range(batch_size)])
        # 将当前时间步的发射矩阵得分与从上一个时间步的标注转移到当前时间步的标注的转移矩阵得分相加
        scores += (emit_score + transition_score) * attention_mask[i]
        transition_to_end = torch.stack([self.transition[tags2label['END_TAG'], 
                                                         tag[attention_mask[:, b].sum().long()]] for b, tag in enumerate(tags.transpose(0, 1))])
        scores += transition_to_end
        return scores

def log_sum_exp(tensor: torch.Tensor, dim: int = -1, keepdim: bool = False) -> torch.Tensor:
    max_score, _ = tensor.max(dim, keepdim=keepdim)
    if keepdim:
        stable_vec = tensor - max_score
    else:
        stable_vec = tensor - max_score.unsqueeze(dim)
    return max_score + (stable_vec.exp().sum(dim, keepdim=keepdim)).log()

def score_all_sentence(self, emit_matrixs, attention_mask):
    seq_len, batch_size, tag_size = emit_matrixs.size()
    alpha = emit_matrixs.new_full((batch_size, tag_size), fill_value=-10000) # [batch_size, tag_size]
    alpha[:, tags2label['START_TAG']] = 0        # 在原始矩阵之前添加一列START_TAG标记
    # 在文本长度维度上进行循环
    for i, emit_score in enumerate(emit_matrixs):
        emit_score = emit_score.unsqueeze(-1)                       # [batch_size, tag_size, 1]
        transition_score = self.transition.unsqueeze(0)             # [1, tag_size, tag_size]
        alpha_score = alpha.unsqueeze(1)                            # [batch_size, 1, tag_size]
        # 这里使用了pytoch中Tensor相加的广播特性，必须思考清楚是如何广播的，以及广播的数据是如何复制的
        alpha_score = alpha_score + transition_score + emit_score   # [batch_size, tag_size, tag_size]
        mask_i = attention_mask[i].unsqueeze(-1)                    # [batch_size, 1] 
        # 通过mask来解决同一个batch_size中不同文本长度的计算问题      
        alpha = log_sum_exp(alpha_score, -1) * mask_i + alpha * (1 - mask_i) # [batch_size, tag_size]
    alpha = alpha + self.transition[tags2label['END_TAG']].unsqueeze(0)

    return log_sum_exp(alpha, -1)

# 解码预测路径
def viterbi_decode(self, emit_matrixs, attention_mask):
    seq_len, batch_size, tag_size = emit_matrixs.size()
    scores = emit_matrixs.new_full((batch_size, tag_size), fill_value=-10000)
    scores[:, tags2label['START_TAG']] = 0
    pointers = []
    for i, emit_score in enumerate(emit_matrixs):
        # 广播机制
        scores_t = scores.unsqueeze(1) + self.transition.unsqueeze(0)  # (batch_size, tag_size, tag_size)
        # max along current_tag to obtain: next_tag score, current_tag pointer
        scores_t, pointer = torch.max(scores_t, -1)  # (batch_size, tag_size), (batch_size, tag_size)
        scores_t += emit_score
        pointers.append(pointer)
        mask_t = attention_mask[i].unsqueeze(-1)  # (batch_size, 1)
        scores = scores_t * mask_t + scores * (1 - mask_t)
        pointers = torch.stack(pointers, 0) # (seq_len, batch_size, tag_size)
        scores += self.transition[tags2label['END_TAG']].unsqueeze(0)
        best_score, best_tag = torch.max(scores, -1)  # (batch_size, ), (batch_size, )
        # 回退
        best_path = best_tag.unsqueeze(-1).tolist() # list shape (batch_size, 1)
        for i in range(batch_size):
            best_tag_i = best_tag[i]
            seq_len_i = int(attention_mask[:, i].sum())
            for ptr_t in reversed(pointers[:seq_len_i, i]):
                # ptr_t shape (tag_size, )
                best_tag_i = ptr_t[best_tag_i].item()
                best_path[i].append(best_tag_i)
                # pop first tag
                best_path[i].pop()
                # reverse order
                best_path[i].reverse()
                return best_path

def prepare_data(path):
    data_dict = {'text': [], 'spo': [], 'label': []}
    real_num_dict = {'部件故障': 0, '性能故障': 0, '检测工具': 0, '组成': 0, '无': 0}
    with open(path, 'r', encoding='utf-8') as f:
        for line in f:
            line = json.loads(line) # 按行处理文件
            entities, real_pair = [], []
            for spo in line['spo_list']: # 获取原始文件中的spo三元组
                data_dict['text'].append(line['text'])
                sub_name, sub_type = spo['h']['name'], relation2type[spo['relation']][0]
                obj_name, obj_type = spo['t']['name'], relation2type[spo['relation']][1]
                entities.append((sub_name, sub_type, spo['h']['pos']))
                entities.append((obj_name, obj_type, spo['t']['pos']))
                data_dict['spo'].append((sub_name, sub_type, obj_name, obj_type, spo['h']['pos'], spo['t']['pos']))
                data_dict['label'].append(relation2label[spo['relation']])
                real_num_dict[spo['relation']] += 1
                real_pair.append((sub_name, sub_type, obj_name, obj_type, spo['h']['pos'], spo['t']['pos']))
            # itertools.combinations()函数:返回entities中所有的2元组合
            all_pairings = list(itertools.combinations(reversed(entities), 2)) + list(itertools.combinations(entities, 2))
           	# 只要每条样本的前20个实体对中构造负样本 
            for pair in all_pairings[: 20]:
                # 判断 当前选择的样本是否为正样本、是否已经被选过
                if (pair[0][0], pair[0][1], pair[1][0], pair[1][1], pair[0][2], pair[1][2]) in real_pair: continue
                # 如果两个实体的类别不能组成关系集合中的结果，也不作为负样本
                if [pair[0][1], pair[1][1]] not in relation2type.values(): continue
                data_dict['text'].append(line['text'])
                data_dict['spo'].append((pair[0][0], pair[0][1], pair[1][0], pair[1][1], pair[0][2], pair[1][2]))
                data_dict['label'].append(relation2label['无'])
                real_num_dict['无'] += 1
                real_pair.append((pair[0][0], pair[0][1], pair[1][0], pair[1][1], pair[0][2], pair[1][2]))
    return pd.DataFrame(data_dict)

    # 这里没有进行特殊处理，直接在文本的510处进行截断

    if self.task != 'test':
        # 遍历每一个三元组
        for spo_dict in self.data.loc[index, 'spo_list']:
            relation_type = spo_dict['relation']

                tag_label[obj_pos + 1: obj_pos + len(obj_name)] = [
                    tags2label[f'I-{obj_type}'] for _ in range(len(obj_name) - 1)
                ]
     return ID, input_ids, attention_mask, tag_label, text

class BertCRF(nn.Module):
    def __init__(self, args):
        super().__init__()
        self.args = args
        self.bert = AutoModel.from_pretrained(args.pretrain_path)
        self.dropout = nn.Dropout(0.3)
        self.fc = nn.Linear(args.hidden_size, len(tags2label))
        self.crf = CRFLayer(args)
        self.score = 0

    def forward(self, input_ids, attention_mask, tags):
        embedding = self.bert(input_ids, attention_mask, return_dict=True)['last_hidden_state']
        emit_matrixs = self.fc(self.dropout(embedding[:, 1: -1, :])).transpose(0, 1)
        attention_mask = attention_mask.transpose(0, 1)[1: -1, :]
        pred_score = self.crf(emit_matrixs, attention_mask)
        gold_score = self.crf.score_sentence(emit_matrixs, tags.transpose(0, 1), attention_mask)
        loss = (pred_score - gold_score).sum()
        return loss
